import numpy as np
import time
import matplotlib
import matplotlib.pyplot as plt
# dfinition des paramtres physiques de l'exprience
Vh = 100           # les cots de la boite sont au potentiel nul
Vb = -100        # potentiel de plaque

# dfinition de la grille de calcul
N = 100                 # nombre de pas sur la grile (identique en Ox et Oy)
V = np.empty([N+1,N+1])     # cration de la grille, en float par dfaut / Va de 0  100 du coup


# critre de prcision du calcul
EPS = 1e-3        # prcision souhaite pour le critre de convergence

# initialisation des compteurs
ecart = 1.0
iteration = 0

# // : division entire

a=N//2
# dfinition des conditions aux limites


# initialisation de l'intrieur de la grille
V[1:N,1:N] = 0.0

# dbut du calcul - enregistrement de la dure
tdebut = time.time()


r=int(np.floor(0.2*N)) # Taille du rayon en pourcentage par rapport  la largeur de la maille
RA=int((N-r)//2) #Partie gauche de vide 
RB=int(RA+r) #Partie droite de vide 
Mid=int(N//2)
#Mid=0


#V[1:-1,1:-1 Va de 1  la derniere valeur -1 en x et pareil en y 

# boucle de calcul - mthode de Gauss-Seidel
while ecart > EPS:
    
    iteration += 1
    # sauvegarde de la grille courante pour calculer l'cart
    Vprec = V.copy()
    # calcul de la nouvelle valeur du potentiel sur la grille
    V[1:-1,1:-1]= 0.25*(V[0:-2,1:-1] +V[2:,1:-1] + V[1:-1,0:-2] + V[1:-1,2:])
    
    # je raffecte les valeurs constantes sur les plaques
    V[0,:]  = Vh  # bord suprieur
    V[a:N+1,0]  = V[a:N+1,1]  # bord gauche inferieur
    V[0:a,0]  = V[0:a,1]  # bord gauche suprieur
    V[a:N+1,100] = V[a:N+1,99]  # bord droit inferieur
    V[0:a,100] = V[0:a,99]   # bord droit suprieur
    #V[0,:] = 0  # bord infrieur
    V[100,0:RA]  = V[99,0:RA]
    V[100,RB:N+1]  = V[99,RB:N+1]
    # critre de convergence
    V[100,40:60]=0
    ecart = np.max(np.abs(V-Vprec))


# Affichage des rsultats
fig = plt.figure(figsize=(14,8))
ax = fig.add_subplot(1,1,1)
